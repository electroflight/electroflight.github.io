ESP32 LED Control

"); client.println("

Turn LED ON

"); client.println("

Turn LED OFF

"); client.println(); break; } else { if (currentLine.indexOf("GET /LED_ON") >= 0) { digitalWrite(ledPin, HIGH); } if (currentLine.indexOf("GET /LED_OFF") >= 0) { digitalWrite(ledPin, LOW); } currentLine = ""; } } else if (c != '\r') { currentLine += c; } } } client.stop(); Serial.println("Client Disconnected."); } }
